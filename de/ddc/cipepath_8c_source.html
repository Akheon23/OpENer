331 const CipDword CipEpathGetLogicalValue(const EipUint8 **message) {
332  LogicalSegmentLogicalFormat logical_format =
333  GetPathLogicalSegmentLogicalFormat(*message);
334  CipDword data = 0;
335  MoveMessageNOctets(1, message); /* Move to logical value */
336  switch (logical_format) {
337  case kLogicalSegmentLogicalFormatEightBit:
338  data = GetSintFromMessage(message);
339  break;
340  case kLogicalSegmentLogicalFormatSixteenBit:
341  MoveMessageNOctets(1, message); /* Pad byte needs to be skipped */
342  data = GetIntFromMessage(message);
343  break;
344  case kLogicalSegmentLogicalFormatThirtyTwoBit:
345  MoveMessageNOctets(1, message); /* Pad byte needs to be skipped */
346  data = GetDintFromMessage(message);
347  break;
348  default:
349  OPENER_ASSERT(false) /* shall not happen! */
350  break;
351  }
352  return data;
353 }
