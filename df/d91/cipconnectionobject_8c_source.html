422 CipUint ConnectionObjectGetRequestedPacketInterval(
423  const CipConnectionObject *const connection_object) {
424  CipUdint remainder_to_resolution =
425  (connection_object->t_to_o_requested_packet_interval) %
426  (kOpenerTimerTickInMilliSeconds * 1000);
427  if( 0 == remainder_to_resolution ) { /* Value can be represented in multiples of the timer resolution */
428  return (CipUint)(connection_object->t_to_o_requested_packet_interval /
429  1000);
430  }
431  else{
432  return (CipUint)(connection_object->t_to_o_requested_packet_interval /
433  1000 - remainder_to_resolution / 1000);
434  }
435 }
