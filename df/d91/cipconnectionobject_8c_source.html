00060   enum {kNodesAmount = OPENER_CIP_NUM_EXPLICIT_CONNS +
00061                        OPENER_CIP_NUM_INPUT_ONLY_CONNS +

00063                        OPENER_CIP_NUM_LISTEN_ONLY_CONNS};
00064   static DoublyLinkedListNode nodes[kNodesAmount];
00065   for(size_t i = 0; i < kNodesAmount; ++i) {
00066     if(nodes[i].previous == NULL && nodes[i].next == NULL && nodes[i].data ==
00067        NULL) {
00068       return &nodes[i];

00378 CipUint ConnectionObjectGetRequestedPacketInterval(
00379   const CipConnectionObject *const connection_object) {
00380   CipUdint remainder_to_resolution =
00381     (connection_object->t_to_o_requested_packet_interval) %
00382     (kOpenerTimerTickInMilliSeconds * 1000);
00383   if( 0 == remainder_to_resolution ) {         /* Value can be represented in multiples of the timer resolution */
00384     return (CipUint)(connection_object->t_to_o_requested_packet_interval /
00385                      1000);
00386   }
00387   else{
00388     return (CipUint)(connection_object->t_to_o_requested_packet_interval /
00389                      1000 - remainder_to_resolution / 1000);
00390   }
00391 }
