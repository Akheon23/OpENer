00076 void ConnectionObjectInitializeFromMessage(
00077   const CipOctet *message,
00078   CipConnectionObject *const
00079   connection_object) {
00080   /* For unconnected send - can be ignored by targets, and is ignored here */
00081   CipByte priority_timetick = GetSintFromMessage(&message);
00082   CipUsint timeout_ticks = GetSintFromMessage(&message);
00083 
00084   /* O_to_T Conn ID */
00085   ConnectionObjectSetCipConsumedConnectionID(connection_object, GetDintFromMessage(&message));
00086   /* T_to_O Conn ID */
00087   ConnectionObjectSetCipProducedConnectionID(connection_object, GetDintFromMessage(&message));
00088 
00089   ConnectionObjectSetConnectionSerialNumber(connection_object, GetIntFromMessage(&message));
00090   ConnectionObjectSetOriginatorVendorId(connection_object, GetIntFromMessage(&message));
00091   ConnectionObjectSetOriginatorSerialNumber(connection_object, GetDintFromMessage(&message));
00092 
00093   /* keep it to none existent till the setup is done this eases error handling and
00094    * the state changes within the forward open request can not be detected from
00095    * the application or from outside (reason we are single threaded)
00096    * */
00097   ConnectionObjectSetState(connection_object, kConnectionObjectStateNonExistent);
00098   connection_object->sequence_count_producing = 0; /* set the sequence count to zero */
00099 
00100   ConnectionObjectSetConnectionTimeoutMultiplier(connection_object, GetSintFromMessage(
00101     &message));
00102 
00103   MoveMessageNOctets(3, &message); /* 3 bytes reserved */
00104 
00105   /* the requested packet interval parameter needs to be a multiple of TIMERTICK from the header file */
00106   OPENER_TRACE_INFO(
00107     "ForwardOpen: ConConnID %" PRIu32 ", ProdConnID %" PRIu32
00108     ", ConnSerNo %u\n",
00109     connection_object->cip_consumed_connection_id,
00110     connection_object->cip_produced_connection_id,
00111     connection_object->connection_serial_number);
00112 
00113   ConnectionObjectSetOToTRequestedPacketInterval(connection_object, GetDintFromMessage(
00114     &message));
00115 
00116   ConnectionObjectResetInactivityWatchdogTimerValue(connection_object);
00117 
00118   //TODO: introduce setter function
00119   connection_object->o_to_t_network_connection_parameters = GetIntFromMessage(&message);
00120 
00121   connection_object->t_to_o_requested_packet_interval = GetDintFromMessage(&message);
00122 
00123   ConnectionObjectSetExpectedPacketRate(connection_object,
00124       connection_object->t_to_o_requested_packet_interval);
00125 
00126   connection_object->t_to_o_network_connection_parameters = GetIntFromMessage(&message);
00127 
00128   connection_object->transport_class_trigger = GetSintFromMessage(&message);
00129 }
