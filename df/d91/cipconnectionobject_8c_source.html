00097 void ConnectionObjectInitializeFromMessage(
00098   const CipOctet **message,
00099   CipConnectionObject *const connection_object) {
00100   /* For unconnected send - can be ignored by targets, and is ignored here */
00101   CipByte priority_timetick = GetSintFromMessage(message);
00102   CipUsint timeout_ticks = GetSintFromMessage(message);
00103 
00104   /* O_to_T Conn ID */
00105   ConnectionObjectSetCipConsumedConnectionID(connection_object, GetDintFromMessage(message));
00106   /* T_to_O Conn ID */
00107   ConnectionObjectSetCipProducedConnectionID(connection_object, GetDintFromMessage(message));
00108 
00109   ConnectionObjectSetConnectionSerialNumber(connection_object, GetIntFromMessage(message));
00110   ConnectionObjectSetOriginatorVendorId(connection_object, GetIntFromMessage(message));
00111   ConnectionObjectSetOriginatorSerialNumber(connection_object, GetDintFromMessage(message));
00112 
00113   /* keep it to none existent till the setup is done this eases error handling and
00114    * the state changes within the forward open request can not be detected from
00115    * the application or from outside (reason we are single threaded)
00116    * */
00117   ConnectionObjectSetState(connection_object, kConnectionObjectStateNonExistent);
00118   connection_object->sequence_count_producing = 0; /* set the sequence count to zero */
00119 
00120   ConnectionObjectSetConnectionTimeoutMultiplier(connection_object, GetSintFromMessage(
00121     message));
00122 
00123   MoveMessageNOctets(3, message); /* 3 bytes reserved */
00124 
00125   /* the requested packet interval parameter needs to be a multiple of TIMERTICK from the header file */
00126   OPENER_TRACE_INFO(
00127     "ForwardOpen: ConConnID %" PRIu32 ", ProdConnID %" PRIu32
00128     ", ConnSerNo %u\n",
00129     connection_object->cip_consumed_connection_id,
00130     connection_object->cip_produced_connection_id,
00131     connection_object->connection_serial_number);
00132 
00133   ConnectionObjectSetOToTRequestedPacketInterval(connection_object, GetDintFromMessage(
00134     message));
00135 
00136   ConnectionObjectSetInitialInactivityWatchdogTimerValue(connection_object);
00137 
00138   //TODO: introduce setter function
00139   connection_object->o_to_t_network_connection_parameters = GetIntFromMessage(message);
00140 
00141   ConnectionObjectSetTToORequestedPacketInterval(connection_object, GetDintFromMessage(message));
00142 
00143   ConnectionObjectSetExpectedPacketRate(connection_object);
00144 
00145   connection_object->t_to_o_network_connection_parameters = GetIntFromMessage(message);
00146 
00147   connection_object->transport_class_trigger = GetSintFromMessage(message);
00148 }

00362 void ConnectionObjectSetExpectedPacketRate(
00363   CipConnectionObject *const connection_object) {
00364         CipUdint remainder_to_resolution = (connection_object->t_to_o_requested_packet_interval) % (kOpenerTimerTickInMilliSeconds * 1000);
00365   if( 0 == remainder_to_resolution ) { /* Value can be represented in multiples of the timer resolution */
00366     connection_object->expected_packet_rate = connection_object->t_to_o_requested_packet_interval / 1000;
00367   }
00368   else{
00369     connection_object->expected_packet_rate = connection_object->t_to_o_requested_packet_interval / 1000 +
00370                                               ((CipUdint)kOpenerTimerTickInMilliSeconds - remainder_to_resolution / 1000);
00371   }
00372 }
