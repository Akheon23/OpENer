389 CipUint ConnectionObjectGetRequestedPacketInterval(
390  const CipConnectionObject *const connection_object) {
391  CipUdint remainder_to_resolution =
392  (connection_object->t_to_o_requested_packet_interval) %
393  (kOpenerTimerTickInMilliSeconds * 1000);
394  if( 0 == remainder_to_resolution ) { /* Value can be represented in multiples of the timer resolution */
395  return (CipUint)(connection_object->t_to_o_requested_packet_interval /
396  1000);
397  }
398  else{
399  return (CipUint)(connection_object->t_to_o_requested_packet_interval /
400  1000 - remainder_to_resolution / 1000);
401  }
402 }
