00685               connection_object->production_inhibit_timer -= elapsed_time;
00686             }
00687           }
00688 
00689           if (connection_object->transmission_trigger_timer <= elapsed_time) { /* need to send package */
00690             OPENER_ASSERT(
00691               NULL != connection_object->connection_send_data_function);
00692             EipStatus eip_status = connection_object
00693                                    ->connection_send_data_function(
00694               connection_object);
00695             if (eip_status == kEipStatusError) {
00696               OPENER_TRACE_ERR(
00697                 "sending of UDP data in manage Connection failed\n");
00698             }
00699             /* reload the timer value */
00700             connection_object->transmission_trigger_timer =
00701               ConnectionObjectGetRequestedPacketInterval(connection_object);
00702             if ( kConnectionObjectTransportClassTriggerProductionTriggerCyclic
00703                  != ConnectionObjectGetTransportClassTriggerProductionTrigger(
00704                    connection_object) ) {
00705               /* non cyclic connections have to reload the production inhibit timer */
00706               ConnectionObjectResetProductionInhibitTimer(connection_object);
00707             }
00708           } else {
00709             connection_object->transmission_trigger_timer -= elapsed_time;
00710           }
00711         }
00712       }
00713     }
00714     node = node->next;
00715   }
00716   return kEipStatusOk;
00717 }

00973 EipStatus CheckElectronicKeyData(
00974   EipUint8 key_format,
00975   void *key_data,
00976   EipUint16 *extended_status
00977   ) {
00978   /* Default return value */
00979   *extended_status = kConnectionManagerExtendedStatusCodeSuccess;
00980 
00981   /* Check key format */
00982   if (4 != key_format) {
00983     *extended_status =
00984       kConnectionManagerExtendedStatusCodeErrorInvalidSegmentTypeInPath;
00985     return kEipStatusError;
00986   }
00987 
00988   bool compatiblity_mode = ElectronicKeyFormat4GetMajorRevisionCompatibility(
00989     key_data);
00990 
00991   /* Check VendorID and ProductCode, must match, or 0 */
00992   if ( ( (ElectronicKeyFormat4GetVendorId(key_data) != vendor_id_) &&
00993          (ElectronicKeyFormat4GetVendorId(key_data) != 0) )
00994        || ( (ElectronicKeyFormat4GetProductCode(key_data) != product_code_)
00995             && (ElectronicKeyFormat4GetProductCode(key_data) != 0) ) ) {
00996     *extended_status =
00997       kConnectionManagerExtendedStatusCodeErrorVendorIdOrProductcodeError;
00998     return kEipStatusError;
00999   } else {
01000     /* VendorID and ProductCode are correct */
01001 
01002     /* Check DeviceType, must match or 0 */
01003     if ( (ElectronicKeyFormat4GetDeviceType(key_data) != device_type_)
01004          && (ElectronicKeyFormat4GetDeviceType(key_data) != 0) ) {
01005       *extended_status =
01006         kConnectionManagerExtendedStatusCodeErrorDeviceTypeError;
01007       return kEipStatusError;
01008     } else {
01009       /* VendorID, ProductCode and DeviceType are correct */
01010 
01011       if (false == compatiblity_mode) {
01012         /* Major = 0 is valid */
01013         if (0 == ElectronicKeyFormat4GetMajorRevision(key_data) ) {
01014           return kEipStatusOk;
01015         }
01016 
01017         /* Check Major / Minor Revision, Major must match, Minor match or 0 */
01018         if ( (ElectronicKeyFormat4GetMajorRevision(key_data) !=
01019               revision_.major_revision)
01020              || ( (ElectronicKeyFormat4GetMinorRevision(key_data) !=
01021                    revision_.minor_revision)
01022                   && (ElectronicKeyFormat4GetMinorRevision(key_data) != 0) ) ) {
01023           *extended_status =
01024             kConnectionManagerExtendedStatusCodeErrorRevisionMismatch;
01025           return kEipStatusError;
01026         }
01027       } else {
01028         /* Compatibility mode is set */
01029 
01030         /* Major must match, Minor != 0 and <= MinorRevision */
01031         if ( (ElectronicKeyFormat4GetMajorRevision(key_data) ==
01032               revision_.major_revision)
01033              && (ElectronicKeyFormat4GetMinorRevision(key_data) > 0)
01034              && (ElectronicKeyFormat4GetMinorRevision(key_data) <=
01035                  revision_.minor_revision) ) {
01036           return kEipStatusOk;
01037         } else {
01038           *extended_status =
01039             kConnectionManagerExtendedStatusCodeErrorRevisionMismatch;
01040           return kEipStatusError;
01041         }
01042       } /* end if CompatiblityMode handling */
01043     }
01044   }
01045 
01046   return
01047     (*extended_status == kConnectionManagerExtendedStatusCodeSuccess) ?
01048     kEipStatusOk : kEipStatusError;
01049 }

01051 EipUint8 ParseConnectionPath(
01052   CipConnectionObject *connection_object,
01053   CipMessageRouterRequest *message_router_request,
01054   EipUint16 *extended_error
01055   ) {
01056   const EipUint8 *message = message_router_request->data;
01057   const size_t connection_path_size = GetSintFromMessage(&message); /* length in words */
01058   size_t remaining_path = connection_path_size;
01059   CipClass *class = NULL;
01060 
01061   CipDword class_id = 0x0;
01062   CipDword instance_id = 0x0;
01063 
01064   /* with 256 we mark that we haven't got a PIT segment */
01065   ConnectionObjectSetProductionInhibitTime(connection_object, 256);
01066 
01067   if ( (g_kForwardOpenHeaderLength + remaining_path * 2)
01068        < message_router_request->request_path_size ) {
01069     /* the received packet is larger than the data in the path */
01070     *extended_error = 0;
01071     return kCipErrorTooMuchData;
01072   }
01073 
01074   if ( (g_kForwardOpenHeaderLength + remaining_path * 2)
01075        > message_router_request->request_path_size ) {
01076     /*there is not enough data in received packet */
01077     *extended_error = 0;
01078     OPENER_TRACE_INFO("Message not long enough for path\n");
01079     return kCipErrorNotEnoughData;
01080   }
01081 
01082   if (remaining_path > 0) {
01083     /* first look if there is an electronic key */
01084     if ( kSegmentTypeLogicalSegment == GetPathSegmentType(message) ) {
01085       if ( kLogicalSegmentLogicalTypeSpecial
01086            == GetPathLogicalSegmentLogicalType(message) ) {
01087         if ( kLogicalSegmentSpecialTypeLogicalFormatElectronicKey
01088              == GetPathLogicalSegmentSpecialTypeLogicalType(message) ) {
01089           if ( kElectronicKeySegmentFormatKeyFormat4
01090                == GetPathLogicalSegmentElectronicKeyFormat(message) ) {
01091             /* Check if there is enough data for holding the electronic key segment */
01092             if (remaining_path < 5) {
01093               *extended_error = 0;
01094               OPENER_TRACE_INFO("Message not long enough for electronic key\n");
01095               return kCipErrorNotEnoughData;
01096             }
01097             /* Electronic key format 4 found */
01098             connection_object->electronic_key.key_format = 4;
01099             ElectronicKeyFormat4 *electronic_key = ElectronicKeyFormat4New();
01100             GetElectronicKeyFormat4FromMessage(&message, electronic_key);
01101             /* logical electronic key found */
01102             connection_object->electronic_key.key_data = electronic_key;
01103 
01104 
01105             remaining_path -= 5; /*length of the electronic key*/
01106             OPENER_TRACE_INFO(
01107               "key: ven ID %d, dev type %d, prod code %d, major %d, minor %d\n",
01108               ElectronicKeyFormat4GetVendorId(connection_object->electronic_key.
01109                                               key_data),
01110               ElectronicKeyFormat4GetDeviceType(connection_object->
01111                                                 electronic_key.key_data),
01112               ElectronicKeyFormat4GetProductCode(connection_object->
01113                                                  electronic_key.key_data),
01114               ElectronicKeyFormat4GetMajorRevision(connection_object->
01115                                                    electronic_key.key_data),
01116               ElectronicKeyFormat4GetMinorRevision(connection_object->
01117                                                    electronic_key.key_data) );
01118             if ( kEipStatusOk
01119                  != CheckElectronicKeyData(
01120                    connection_object->electronic_key.key_format,
01121                    connection_object->electronic_key.key_data,
01122                    extended_error) ) {
01123               ElectronicKeyFormat4Delete(&electronic_key);
01124               return kCipErrorConnectionFailure;
01125             }
01126             ElectronicKeyFormat4Delete(&electronic_key);
01127           }
01128 
01129         } else {
01130           OPENER_TRACE_INFO("no key\n");
01131         }
01132       }
01133     }
01134 
01135     //TODO: Refactor this afterwards
01136     if ( kConnectionObjectTransportClassTriggerProductionTriggerCyclic
01137          != ConnectionObjectGetTransportClassTriggerProductionTrigger(
01138            connection_object) ) {
01139       /*non cyclic connections may have a production inhibit */
01140       if ( kSegmentTypeNetworkSegment == GetPathSegmentType(message) ) {
01141         NetworkSegmentSubtype network_segment_subtype =
01142           GetPathNetworkSegmentSubtype(message);
01143         if (kNetworkSegmentSubtypeProductionInhibitTimeInMilliseconds
01144             == network_segment_subtype) {
01145           OPENER_TRACE_INFO("PIT segment available - value: %u\n",message[1]);
01146           connection_object->production_inhibit_time = message[1];
01147           message += 2;
01148           remaining_path -= 1;
01149         }
01150       }
01151     }
01152 
01153     if (kSegmentTypeLogicalSegment == GetPathSegmentType(message) &&
01154         kLogicalSegmentLogicalTypeClassId ==
01155         GetPathLogicalSegmentLogicalType(message) ) {
01156 
01157       class_id = CipEpathGetLogicalValue(&message);
01158       class = GetCipClass(class_id);
01159       if (NULL == class) {
01160         OPENER_TRACE_ERR("classid %" PRIx32 " not found\n",
01161                          class_id);
01162 
01163         if (class_id >= 0xC8) { /*reserved range of class ids */
01164           *extended_error =
01165             kConnectionManagerExtendedStatusCodeErrorInvalidSegmentTypeInPath;
01166         } else {
01167           *extended_error =
01168             kConnectionManagerExtendedStatusCodeInconsistentApplicationPathCombo;
01169         }
01170         return kCipErrorConnectionFailure;
01171       }
01172 
01173       OPENER_TRACE_INFO("classid %" PRIx32 " (%s)\n",
01174                         class_id,
01175                         class->class_name);
01176     } else {
01177       *extended_error =
01178         kConnectionManagerExtendedStatusCodeErrorInvalidSegmentTypeInPath;
01179       return kCipErrorConnectionFailure;
01180     }
01181     remaining_path -= 1; /* 1 16Bit word for the class part of the path */
01182 
01183     /* Get instance ID */
01184     if ( kSegmentTypeLogicalSegment == GetPathSegmentType(message) &&
01185          kLogicalSegmentLogicalTypeInstanceId ==
01186          GetPathLogicalSegmentLogicalType(message) ) {                                                                                                       /* store the configuration ID for later checking in the application connection types */
01187       instance_id = CipEpathGetLogicalValue(&message);
01188 
01189       OPENER_TRACE_INFO("Configuration instance id %" PRId32 "\n",
01190                         instance_id);
01191       if ( NULL == GetCipInstance(class, instance_id) ) {
01192         /*according to the test tool we should respond with this extended error code */
01193         *extended_error =
01194           kConnectionManagerExtendedStatusCodeErrorInvalidSegmentTypeInPath;
01195         return kCipErrorConnectionFailure;
01196       }
01197       /* 1 or 2 16Bit words for the configuration instance part of the path  */
01198       remaining_path -= (instance_id > 0xFF) ? 2 : 1; //TODO: 32 bit case missing
01199     } else {
01200       OPENER_TRACE_INFO("no config data\n");
01201     }
01202 
01203     if ( kConnectionObjectTransportClassTriggerTransportClass3 ==
01204          ConnectionObjectGetTransportClassTriggerTransportClass(
01205            connection_object) )
01206     {
01207       /*we have Class 3 connection*/
01208       if (remaining_path > 0) {
01209         OPENER_TRACE_WARN(
01210           "Too much data in connection path for class 3 connection\n");
01211         *extended_error =
01212           kConnectionManagerExtendedStatusCodeErrorInvalidSegmentTypeInPath;
01213         return kCipErrorConnectionFailure;
01214       }
01215 
01216       /* connection end point has to be the message router instance 1 */
01217       if ( (class_id != kCipMessageRouterClassCode)
01218            || (1 != instance_id) ) {
01219         *extended_error =
01220           kConnectionManagerExtendedStatusCodeInconsistentApplicationPathCombo;
01221         return kCipErrorConnectionFailure;
01222       }
01223       /* Configuration connection point is producing connection point */
01224       CipConnectionPathEpath connection_epath = {
01225         .class_id = class_id,
01226         .instance_id = instance_id,
01227         .attribute_id_or_connection_point = 0
01228       };
01229 
01230       memcpy(&(connection_object->configuration_path),
01231              &connection_epath,
01232              sizeof(connection_object->configuration_path) );
01233       memcpy(&(connection_object->produced_path), &connection_epath,
01234              sizeof(connection_object->produced_path) );
01235 
01236       /* End class 3 connection handling */
01237     } else { /* we have an IO connection */
01238       //TODO: Check in old implementation how this was done (this is a fix for checking)
01239       CipConnectionPathEpath connection_epath = {
01240         .class_id = class_id,
01241         .instance_id = instance_id,
01242         .attribute_id_or_connection_point = 0
01243       };
01244       memcpy(&(connection_object->configuration_path),
01245              &connection_epath,
01246              sizeof(connection_object->configuration_path) );
01247       ConnectionObjectConnectionType originator_to_target_connection_type =
01248         ConnectionObjectGetOToTConnectionType(
01249           connection_object);
01250       ConnectionObjectConnectionType target_to_originator_connection_type =
01251         ConnectionObjectGetTToOConnectionType(
01252           connection_object);
01253 
01254       connection_object->consumed_connection_path_length = 0;
01255       connection_object->consumed_connection_path = NULL;
01256       //connection_object->connection_path.connection_point[1] = 0; /* set not available path to Invalid */
01257 
01258       size_t number_of_encoded_paths = 0;
01259       CipConnectionPathEpath *paths_to_encode[2] = { 0 };
01260       if (kConnectionObjectConnectionTypeNull ==
01261           originator_to_target_connection_type) {
01262         if (kConnectionObjectConnectionTypeNull ==
01263             target_to_originator_connection_type) {                                        /* configuration only connection */
01264           number_of_encoded_paths = 0;
01265           OPENER_TRACE_WARN("assembly: type invalid\n");
01266         } else { /* 1 path -> path is for production */
01267           OPENER_TRACE_INFO("assembly: type produce\n");
01268           number_of_encoded_paths = 1;
01269           paths_to_encode[0] = &(connection_object->produced_path);
01270         }
01271       } else {
01272         if (kConnectionObjectConnectionTypeNull ==
01273             target_to_originator_connection_type) {                                        /* 1 path -> path is for consumption */
01274           OPENER_TRACE_INFO("assembly: type consume\n");
01275           number_of_encoded_paths = 1;
01276           paths_to_encode[0] = &(connection_object->consumed_path);
01277         } else { /* 2 paths -> 1st for production 2nd for consumption */
01278           OPENER_TRACE_INFO("assembly: type bidirectional\n");
01279           paths_to_encode[0] = &(connection_object->consumed_path);
01280           paths_to_encode[1] = &(connection_object->produced_path);
01281           number_of_encoded_paths = 2;
01282         }
01283       }
01284 
01285       for (size_t i = 0; i < number_of_encoded_paths; i++) /* process up to 2 encoded paths */
01286       {

01289              kLogicalSegmentLogicalTypeConnectionPoint ==

01294             .class_id = class_id,
01295             .instance_id = attribute_id,
01296             .attribute_id_or_connection_point = 0
01297           };
01298           memcpy(paths_to_encode[i], &connection_epath,
01299                  sizeof(connection_object->produced_path) );
01300           OPENER_TRACE_INFO(
01301             "connection point %" PRIu32 "\n",
01302             attribute_id);
01303           if ( NULL
01304                == GetCipInstance(
01305                  class,
01306                  attribute_id) ) { /* Old code - Probably here the attribute ID marks the instance for the assembly object  */

01308               kConnectionManagerExtendedStatusCodeInconsistentApplicationPathCombo;
01309             return kCipErrorConnectionFailure;
01310           }
01311           /* 1 or 2 16Bit word for the connection point part of the path */
01312           remaining_path -= (attribute_id > 0xFF) ? 2 : 1;
01313         } else {

01315             kConnectionManagerExtendedStatusCodeErrorInvalidSegmentTypeInPath;
01316           return kCipErrorConnectionFailure;
01317         }
01318       }

01321       g_config_data_buffer = NULL;

01329               message);
01330             switch (data_segment_type) {
01331               case kDataSegmentSubtypeSimpleData:
01332                 g_config_data_length = message[1] * 2; /*data segments store length 16-bit word wise */
01333                 g_config_data_buffer = (EipUint8 *) message + 2;
01334                 remaining_path -= (g_config_data_length + 2) / 2;
01335                 message += (g_config_data_length + 2);
01336                 break;
01337               default:
01338                 OPENER_TRACE_ERR("Not allowed in connection manager");
01339                 break;
01340             }
01341           }
01342           break;
01343           case kSegmentTypeNetworkSegment: {
01344             NetworkSegmentSubtype subtype = GetPathNetworkSegmentSubtype(
01345               message);
01346             switch (subtype) {
01347               case kNetworkSegmentSubtypeProductionInhibitTimeInMilliseconds:
01348                 if (
01349                   kConnectionObjectTransportClassTriggerProductionTriggerCyclic
01350                   !=
01351                   ConnectionObjectGetTransportClassTriggerProductionTrigger(
01352                     connection_object) ) {
01353                   /* only non cyclic connections may have a production inhibit */
01354                   connection_object->production_inhibit_time = message[1];
01355                   message += 2;
01356                   remaining_path -= 2;
01357                 } else {
01358                   *extended_error = connection_path_size - remaining_path; /*offset in 16Bit words where within the connection path the error happend*/
01359                   return kCipErrorPathSegmentError; /*status code for invalid segment type*/
01360                 }
01361               default:
01362                 OPENER_TRACE_ERR("Not allowed in connection manager");
01363                 break;
01364             }
01365           }
01366           break;
01367 
01368           default:
01369             OPENER_TRACE_WARN(
01370               "No data segment identifier found for the configuration data\n");
01371             *extended_error = connection_path_size - remaining_path; /*offset in 16Bit words where within the connection path the error happend*/
01372             return
01373               kConnectionManagerGeneralStatusPathSegmentErrorInUnconnectedSend;
01374             break;
01375         }
01376       }
01377     }
01378   }
01379 
01380   OPENER_TRACE_INFO("Resulting PIT value: %u\n",
01381                     connection_object->production_inhibit_time);

01383   message_router_request->data = message;
01384   return kEipStatusOk;
