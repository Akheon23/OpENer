648 EipStatus ManageConnections(MilliSeconds elapsed_time) {
649  //OPENER_TRACE_INFO("Entering ManageConnections\n");
650  /*Inform application that it can execute */
651  HandleApplication();
652  ManageEncapsulationMessages(elapsed_time);

655 
656  while (NULL != node) {
657  //OPENER_TRACE_INFO("Entering Connection Object loop\n");
658  CipConnectionObject *connection_object = node->data;
659  if (kConnectionObjectStateEstablished ==
660  ConnectionObjectGetState(connection_object) ) {
661  if ( (NULL != connection_object->consuming_instance) || /* we have a consuming connection check inactivity watchdog timer */
662  (kConnectionObjectTransportClassTriggerDirectionServer ==
663  ConnectionObjectGetTransportClassTriggerDirection(connection_object) ) ) /* all server connections have to maintain an inactivity watchdog timer */
664  {
665  if (elapsed_time >= connection_object->inactivity_watchdog_timer) {
666  /* we have a timed out connection perform watchdog time out action*/
667  OPENER_TRACE_INFO(">>>>>>>>>>Connection ConnNr: %u timed out\n",

670  connection_object->connection_timeout_function(connection_object);
671  } else {
672  connection_object->inactivity_watchdog_timer -= elapsed_time;
673  connection_object->last_package_watchdog_timer -= elapsed_time;
674  }
675  }
676  /* only if the connection has not timed out check if data is to be send */
677  if (kConnectionObjectStateEstablished ==
678  ConnectionObjectGetState(connection_object) ) {
679  /* client connection */
680  if ( (0 != ConnectionObjectGetExpectedPacketRate(connection_object) )
681  && (kEipInvalidSocket

687  != ConnectionObjectGetTransportClassTriggerProductionTrigger(
688  connection_object) ) {

690  if (elapsed_time <= connection_object->production_inhibit_timer) {
691  //The connection is allowed to send again
692  } else {
693  connection_object->production_inhibit_timer -= elapsed_time;
694  }
695  }

718  }
719  }
