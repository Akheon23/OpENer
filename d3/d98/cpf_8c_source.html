00594 int AssembleLinearMessage(

00596   CipCommonPacketFormatData *common_packet_format_data_item,
00597   EipUint8 *message) {
00598 
00599   size_t message_size = 0;
00600 
00601   if (message_router_response) {
00602     /* add Interface Handle and Timeout = 0 -> only for SendRRData and SendUnitData necessary */
00603     AddDintToMessage(0, &message);
00604     AddIntToMessage(0, &message);
00605     message_size += 6;
00606   }
00607 
00608   message_size = EncodeItemCount(common_packet_format_data_item, &message,
00609                                  message_size);
00610 
00611   /* process Address Item */
00612   switch (common_packet_format_data_item->address_item.type_id) {
00613     case kCipItemIdNullAddress: {
00614       message_size = EncodeNullAddressItem(&message, message_size);
00615       break;
00616     }
00617     case kCipItemIdConnectionAddress: {
00618       message_size = EncodeConnectedAddressItem(&message,
00619                                                 common_packet_format_data_item,
00620                                                 message_size);
00621       break;
00622     }
00623     case kCipItemIdSequencedAddressItem: {
00624       message_size = EncodeSequencedAddressItem(&message,
00625                                                 common_packet_format_data_item,
00626                                                 message_size);
00627       break;
00628     }
00629   }
00630 
00631   /* process Data Item */
00632   if ( (common_packet_format_data_item->data_item.type_id
00633         == kCipItemIdUnconnectedDataItem)
00634        || (common_packet_format_data_item->data_item.type_id
00635            == kCipItemIdConnectedDataItem) ) {
00636 
00637     if (message_router_response) {
00638       message_size = EncodeDataItemType(common_packet_format_data_item,
00639                                         &message, message_size);
00640 
00641       if (common_packet_format_data_item->data_item.type_id
00642           == kCipItemIdConnectedDataItem) { /* Connected Item */
00643         message_size = EncodeConnectedDataItemLength(message_router_response,
00644                                                      &message, message_size);
00645         message_size = EncodeSequenceNumber(message_size,
00646                                             &g_common_packet_format_data_item,
00647                                             &message);
00648 
00649       } else { /* Unconnected Item */
00650         message_size = EncodeUnconnectedDataItemLength(message_size,
00651                                                        message_router_response,
00652                                                        &message);
00653       }
00654 
00655       /* write message router response into linear memory */
00656       message_size = EncodeReplyService(message_size, &message,
00657                                         message_router_response);
00658       message_size = EncodeReservedFieldOfLengthByte(message_size, &message,
00659                                                      message_router_response);
00660       message_size = EncodeGeneralStatus(message_size, &message,
00661                                          message_router_response);
00662       message_size = EncodeExtendedStatus(message_size, &message,
00663                                           message_router_response);
00664       message_size = EncodeMessageRouterResponseData(message_size,
00665                                                      message_router_response,
00666                                                      &message);
00667     } else { /* connected IO Message to send */
00668       message_size = EncodeDataItemType(common_packet_format_data_item,
00669                                         &message, message_size);
00670 
00671       message_size = EncodeDataItemLength(common_packet_format_data_item,
00672                                           &message, message_size);
00673 
00674       message_size = EncodeDataItemData(common_packet_format_data_item,
00675                                         &message, message_size);

00679   /* process SockAddr Info Items */
00680   /* make sure first the O->T and then T->O appears on the wire.
00681    * EtherNet/IP specification doesn't demand it, but there are EIP
00682    * devices which depend on CPF items to appear in the order of their

00685        type <= kCipItemIdSocketAddressInfoTargetToOriginator; type++) {

00693                                                 common_packet_format_data_item,
00694                                                 &message);
00695

00697           common_packet_format_data_item->address_info_item[j].sin_port,
00698           common_packet_format_data_item->address_info_item[j].sin_addr,
00699           &message);
00700

00702           0, 8, &message);

00704       }
00705     }
