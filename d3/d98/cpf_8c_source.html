135 EipStatus CreateCommonPacketFormatStructure(
136  const EipUint8 *data,
137  size_t data_length,
138  CipCommonPacketFormatData *common_packet_format_data) {
139 
140  common_packet_format_data->address_info_item[0].type_id = 0;
141  common_packet_format_data->address_info_item[1].type_id = 0;
142 
143  int length_count = 0;
144  common_packet_format_data->item_count = GetIntFromMessage(&data);
145  length_count += 2;
146  if (common_packet_format_data->item_count >= 1) {
147  common_packet_format_data->address_item.type_id = GetIntFromMessage(&data);
148  common_packet_format_data->address_item.length = GetIntFromMessage(&data);
149  length_count += 4;
150  if (common_packet_format_data->address_item.length >= 4) {
151  common_packet_format_data->address_item.data.connection_identifier =
152  GetDintFromMessage(&data);
153  length_count += 4;
154  }
155  if (common_packet_format_data->address_item.length == 8) {
156  common_packet_format_data->address_item.data.sequence_number =
157  GetDintFromMessage(&data);
158  length_count += 4;
159  }
160  }
161  if (common_packet_format_data->item_count >= 2) {
162  common_packet_format_data->data_item.type_id = GetIntFromMessage(&data);
163  common_packet_format_data->data_item.length = GetIntFromMessage(&data);
164  common_packet_format_data->data_item.data = (EipUint8 *)data;
165  data += common_packet_format_data->data_item.length;
166  length_count += (4 + common_packet_format_data->data_item.length);
167  }
168  for (int j = 0; j < (common_packet_format_data->item_count - 2); j++) /* TODO there needs to be a limit check here???*/
169  {
170  common_packet_format_data->address_info_item[j].type_id = GetIntFromMessage(
171  &data);
172  OPENER_TRACE_INFO("Sockaddr type id: %x\n",
173  common_packet_format_data->address_info_item[j].type_id);
174  length_count += 2;
175  if ( (common_packet_format_data->address_info_item[j].type_id
176  == kCipItemIdSocketAddressInfoOriginatorToTarget)
177  || (common_packet_format_data->address_info_item[j].type_id
178  == kCipItemIdSocketAddressInfoTargetToOriginator) ) {
179  common_packet_format_data->address_info_item[j].length =
180  GetIntFromMessage(&data);
181  common_packet_format_data->address_info_item[j].sin_family =
182  GetIntFromMessage(&data);
183  common_packet_format_data->address_info_item[j].sin_port =
184  GetIntFromMessage(&data);
185  common_packet_format_data->address_info_item[j].sin_addr =
186  GetDintFromMessage(&data);
187  for (int i = 0; i < 8; i++) {
188  common_packet_format_data->address_info_item[j].nasin_zero[i] = *data;
189  data++;
190  }
191  length_count += 18;
192  } else { /* no sockaddr item found */
193  common_packet_format_data->address_info_item[j].type_id = 0; /* mark as not set */
194  data -= 2;
195  }
196  }
197  /* set the addressInfoItems to not set if they were not received */
198  if (common_packet_format_data->item_count < 4) {
199  common_packet_format_data->address_info_item[1].type_id = 0;
200  if (common_packet_format_data->item_count < 3) {
201  common_packet_format_data->address_info_item[0].type_id = 0;
202  }
203  }
204  if (length_count == data_length) { /* length of data is equal to length of Addr and length of Data */
205  return kEipStatusOk;
206  } else {
207  OPENER_TRACE_WARN(
208  "something is wrong with the length in Message Router @ CreateCommonPacketFormatStructure\n");
209  if (common_packet_format_data->item_count > 2) {
210  /* there is an optional packet in data stream which is not sockaddr item */
211  return kEipStatusOk;
212  } else { /* something with the length was wrong */
213  return kEipStatusError;
214  }
215  }
216 }

595 int AssembleLinearMessage(
596  CipMessageRouterResponse *message_router_response,
597  CipCommonPacketFormatData *common_packet_format_data_item,
598  EipUint8 *message) {
599 
600  size_t message_size = 0;
601 
602  if (message_router_response) {
603  /* add Interface Handle and Timeout = 0 -> only for SendRRData and SendUnitData necessary */
604  AddDintToMessage(0, &message);
605  AddIntToMessage(0, &message);
606  message_size += 6;
607  }
608 
609  message_size = EncodeItemCount(common_packet_format_data_item, &message,
610  message_size);
611 
612  /* process Address Item */
613  switch (common_packet_format_data_item->address_item.type_id) {
614  case kCipItemIdNullAddress: {
615  message_size = EncodeNullAddressItem(&message, message_size);
616  break;
617  }
618  case kCipItemIdConnectionAddress: {
619  message_size = EncodeConnectedAddressItem(&message,
620  common_packet_format_data_item,
621  message_size);
622  break;
623  }
624  case kCipItemIdSequencedAddressItem: {
625  message_size = EncodeSequencedAddressItem(&message,
626  common_packet_format_data_item,
627  message_size);
628  break;
629  }
630  }
631 
632  /* process Data Item */
633  if ( (common_packet_format_data_item->data_item.type_id
634  == kCipItemIdUnconnectedDataItem)
635  || (common_packet_format_data_item->data_item.type_id
636  == kCipItemIdConnectedDataItem) ) {
637 
638  if (message_router_response) {
639  message_size = EncodeDataItemType(common_packet_format_data_item,
640  &message, message_size);
641 
642  if (common_packet_format_data_item->data_item.type_id
643  == kCipItemIdConnectedDataItem) { /* Connected Item */
644  message_size = EncodeConnectedDataItemLength(message_router_response,
645  &message, message_size);
646  message_size = EncodeSequenceNumber(message_size,
647  &g_common_packet_format_data_item,
648  &message);
649 
650  } else { /* Unconnected Item */
651  message_size = EncodeUnconnectedDataItemLength(message_size,
652  message_router_response,
653  &message);
654  }
655 
656  /* write message router response into linear memory */
657  message_size = EncodeReplyService(message_size, &message,
658  message_router_response);
659  message_size = EncodeReservedFieldOfLengthByte(message_size, &message,
660  message_router_response);
661  message_size = EncodeGeneralStatus(message_size, &message,
662  message_router_response);
663  message_size = EncodeExtendedStatus(message_size, &message,
664  message_router_response);
665  message_size = EncodeMessageRouterResponseData(message_size,
666  message_router_response,
667  &message);
668  } else { /* connected IO Message to send */
669  message_size = EncodeDataItemType(common_packet_format_data_item,
670  &message, message_size);
671 
672  message_size = EncodeDataItemLength(common_packet_format_data_item,
673  &message, message_size);
674 
675  message_size = EncodeDataItemData(common_packet_format_data_item,
676  &message, message_size);
677  }
678  }
679 
680  /* process SockAddr Info Items */
681  /* make sure first the O->T and then T->O appears on the wire.
682  * EtherNet/IP specification doesn't demand it, but there are EIP
683  * devices which depend on CPF items to appear in the order of their
684  * ID number */
685  for (int type = kCipItemIdSocketAddressInfoOriginatorToTarget;
686  type <= kCipItemIdSocketAddressInfoTargetToOriginator; type++) {
687  for (int j = 0; j < 2; j++) {
688  if (common_packet_format_data_item->address_info_item[j].type_id
689  == type) {
690  message_size = EncodeSockaddrInfoItemTypeId(
691  message_size, j, common_packet_format_data_item, &message);
692 
693  message_size = EncodeSockaddrInfoLength(message_size, j,
694  common_packet_format_data_item,
695  &message);
696 
697  message_size += EncapsulateIpAddress(
698  common_packet_format_data_item->address_info_item[j].sin_port,
699  common_packet_format_data_item->address_info_item[j].sin_addr,
700  &message);
701 
702  message_size += FillNextNMessageOctetsWithValueAndMoveToNextPosition(
703  0, 8, &message);
704  break;
705  }
706  }
707  }
708  return message_size;
709 }
