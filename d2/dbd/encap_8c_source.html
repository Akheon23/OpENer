162 int HandleReceivedExplictTcpData(int socket,
163  EipUint8 *buffer,
164  size_t length,
165  int *remaining_bytes,
166  struct sockaddr *originator_address) {
167  OPENER_TRACE_INFO("Handles data for TCP socket: %d\n", socket);
168  EipStatus return_value = kEipStatusOk;
169  EncapsulationData encapsulation_data = {0};
170  /* eat the encapsulation header*/
171  /* the structure contains a pointer to the encapsulated data*/
172  /* returns how many bytes are left after the encapsulated data*/
173  *remaining_bytes = CreateEncapsulationStructure(buffer, length,
174  &encapsulation_data);
175 
176  if (kEncapsulationHeaderOptionsFlag == encapsulation_data.options) /*TODO generate appropriate error response*/
177  {
178  if (*remaining_bytes >= 0) /* check if the message is corrupt: header size + claimed payload size > than what we actually received*/
179  {
180  /* full package or more received */
181  encapsulation_data.status = kEncapsulationProtocolSuccess;
182  return_value = kEipStatusOkSend;
183  /* most of these functions need a reply to be send */
184  switch (encapsulation_data.command_code) {
185  case (kEncapsulationCommandNoOperation):
186  OPENER_TRACE_INFO("NOP\n");
187  /* NOP needs no reply and does nothing */
188  return_value = kEipStatusOk;
189  break;
190 
191  case (kEncapsulationCommandListServices):
192  OPENER_TRACE_INFO("List services\n");
193  HandleReceivedListServicesCommand(&encapsulation_data);
194  break;
195 
196  case (kEncapsulationCommandListIdentity):
197  OPENER_TRACE_INFO("List identity\n");
198  HandleReceivedListIdentityCommandTcp(&encapsulation_data);
199  break;
200 
201  case (kEncapsulationCommandListInterfaces):
202  OPENER_TRACE_INFO("List interfaces\n");
203  HandleReceivedListInterfacesCommand(&encapsulation_data);
204  break;
205 
206  case (kEncapsulationCommandRegisterSession):
207  OPENER_TRACE_INFO("Register session\n");
208  HandleReceivedRegisterSessionCommand(socket, &encapsulation_data);
209  break;
210 
211  case (kEncapsulationCommandUnregisterSession):
212  OPENER_TRACE_INFO("unregister session\n");
213  return_value = HandleReceivedUnregisterSessionCommand(
214  &encapsulation_data);
215  break;
216 
217  case (kEncapsulationCommandSendRequestReplyData):
218  OPENER_TRACE_INFO("Send Request/Reply Data\n");
219  return_value = HandleReceivedSendRequestResponseDataCommand(
220  &encapsulation_data, originator_address);
221  break;
222 
223  case (kEncapsulationCommandSendUnitData):
224  OPENER_TRACE_INFO("Send Unit Data\n");
225  return_value = HandleReceivedSendUnitDataCommand(&encapsulation_data,
226  originator_address);
227  break;
228 
229  default:
230  encapsulation_data.status = kEncapsulationProtocolInvalidCommand;
231  encapsulation_data.data_length = 0;
232  break;
233  }
234  /* if nRetVal is greater than 0 data has to be sent */
235  if (kEipStatusOk < return_value) {
236  return_value = EncapsulateData(&encapsulation_data);
237  }
238  }
239  }
240 
241  return return_value;
242 }

392 ptrdiff_t EncapsulateListIdentyResponseMessage(
393  EipByte *const communication_buffer) {
394  CipOctet *communication_buffer_runner = communication_buffer;
395 
396  AddIntToMessage( 1, &(communication_buffer_runner) ); /* Item count: one item */
397  AddIntToMessage(kCipItemIdListIdentityResponse, &communication_buffer_runner);
398 
399  EipByte *id_length_buffer = communication_buffer_runner;
400  MoveMessageNOctets(2, (const CipOctet **)&communication_buffer_runner); /*at this place the real length will be inserted below*/
401 
402  AddIntToMessage(kSupportedProtocolVersion, &communication_buffer_runner);
403 
404  EncapsulateIpAddress(htons(kOpenerEthernetPort),
405  interface_configuration_.ip_address,
406  &communication_buffer_runner);
407 
408  memset(communication_buffer_runner, 0, 8);
409  MoveMessageNOctets(8, (const CipOctet **)&communication_buffer_runner);
410 
411  AddIntToMessage(vendor_id_, &communication_buffer_runner);
412  AddIntToMessage(device_type_, &communication_buffer_runner);
413  AddIntToMessage(product_code_, &communication_buffer_runner);
414  *(communication_buffer_runner)++ = revision_.major_revision;
415  *(communication_buffer_runner)++ = revision_.minor_revision;
416  AddIntToMessage(status_, &communication_buffer_runner);
417  AddDintToMessage(serial_number_, &communication_buffer_runner);
418  *communication_buffer_runner++ = (unsigned char) product_name_.length;
419  memcpy(communication_buffer_runner, product_name_.string,
420  product_name_.length);
421  communication_buffer_runner += product_name_.length;
422  *communication_buffer_runner++ = 0xFF;
423 
424  AddIntToMessage(communication_buffer_runner - id_length_buffer - 2,
425  &id_length_buffer); /* the -2 is for not counting the length field*/
426 
427  return communication_buffer_runner - communication_buffer;
428 }

433  MoveMessageNOctets(12, (const CipOctet **)&buffer_start); /* start of the sender context */

445 /* @brief Check supported protocol, generate session handle, send replay back to originator.
446  * @param socket Socket this request is associated to. Needed for double register check
447  * @param receive_data Pointer to received data with request/response.
448  */
449 void HandleReceivedRegisterSessionCommand(int socket,
450  EncapsulationData *receive_data) {
451  int session_index = 0;
452  const EipUint8 *receive_data_buffer = NULL;
453  EipUint16 protocol_version = GetIntFromMessage(
454  (const EipUint8 **const)&receive_data->
455  current_communication_buffer_position);
456  EipUint16 nOptionFlag = GetIntFromMessage(
457  (const EipUint8 **const)&receive_data->
458  current_communication_buffer_position);
459 
460  /* check if requested protocol version is supported and the register session option flag is zero*/
461  if ( (0 < protocol_version) && (protocol_version <= kSupportedProtocolVersion)
462  && (0 == nOptionFlag) ) { /*Option field should be zero*/
463  /* check if the socket has already a session open */
464  for (int i = 0; i < OPENER_NUMBER_OF_SUPPORTED_SESSIONS; ++i) {
465  if (g_registered_sessions[i] == socket) {
466  /* the socket has already registered a session this is not allowed*/
467  OPENER_TRACE_INFO(
468  "Error: A session is already registered at socket %d\n",
469  socket);
470  receive_data->session_handle = i + 1; /*return the already assigned session back, the cip spec is not clear about this needs to be tested*/
471  receive_data->status = kEncapsulationProtocolInvalidCommand;
472  session_index = kSessionStatusInvalid;
473  receive_data_buffer =
474  &receive_data->communication_buffer_start[
475  kEncapsulationHeaderSessionHandlePosition];
476  AddDintToMessage(receive_data->session_handle,
477  (EipUint8 **const)&receive_data_buffer); /*EncapsulateData will not update the session handle so we have to do it here by hand*/
478  break;
479  }
480  }
481 
482  if (kSessionStatusInvalid != session_index) {
483  session_index = GetFreeSessionIndex();
484  if (kSessionStatusInvalid == session_index) /* no more sessions available */
485  {
486  receive_data->status = kEncapsulationProtocolInsufficientMemory;
487  } else { /* successful session registered */
488  SocketTimer *socket_timer = SocketTimerArrayGetEmptySocketTimer(
489  g_timestamps,
490  OPENER_NUMBER_OF_SUPPORTED_SESSIONS);
491  SocketTimerSetSocket(socket_timer, socket);
492  SocketTimerSetLastUpdate(socket_timer, g_actual_time);
493  g_registered_sessions[session_index] = socket; /* store associated socket */
494  receive_data->session_handle = session_index + 1;
495  receive_data->status = kEncapsulationProtocolSuccess;
496  receive_data_buffer =
497  &receive_data->communication_buffer_start[
498  kEncapsulationHeaderSessionHandlePosition];
499  AddDintToMessage(receive_data->session_handle,
500  (EipUint8 **const)&receive_data_buffer); /*EncapsulateData will not update the session handle so we have to do it here by hand*/
501  }
502  }
503  } else { /* protocol not supported */
504  receive_data->status = kEncapsulationProtocolUnsupportedProtocol;
505  }
506 
507  receive_data->data_length = 4;
508 }

541  struct sockaddr *originator_address)
542 {
543  EipStatus return_value = kEipStatusOkSend;
544 
545  if (receive_data->data_length >= 6) {
546  /* Command specific data UDINT .. Interface Handle, UINT .. Timeout, CPF packets */
547  /* don't use the data yet */
548  GetDintFromMessage(
549  (const EipUint8 **const)&receive_data->
550  current_communication_buffer_position ); /* skip over null interface handle*/
551  GetIntFromMessage(
552  (const EipUint8 **const)&receive_data->
553  current_communication_buffer_position ); /* skip over unused timeout value*/
554  receive_data->data_length -= 6; /* the rest is in CPF format*/
555

557  {

562  ],
563  originator_address);
564

566  receive_data->data_length = send_size;
567  } else {
568  return_value = kEipStatusError;
569  }
570  } else { /* received a package with non registered session handle */
571  receive_data->data_length = 0;
572  receive_data->status = kEncapsulationProtocolInvalidSessionHandle;
573  }
