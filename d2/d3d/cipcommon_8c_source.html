126 CipInstance *AddCipInstances(CipClass *RESTRICT const cip_class,
127  const int number_of_instances) {
128  CipInstance **next_instance = NULL;
129  EipUint32 instance_number = 1; /* the first instance is number 1 */
130 
131  OPENER_TRACE_INFO("adding %d instances to class %s\n", number_of_instances,
132  cip_class->class_name);
133 
134  next_instance = &cip_class->instances; /* get address of pointer to head of chain */
135  while (*next_instance) /* as long as what pp points to is not zero */
136  {
137  next_instance = &(*next_instance)->next; /* follow the chain until pp points to pointer that contains a zero */
138  instance_number++; /* keep track of what the first new instance number will be */
139  }
140 
141  CipInstance *current_instance = current_instance = (CipInstance *) CipCalloc(
142  number_of_instances, sizeof(CipInstance) ); /* allocate a block of memory for all created instances*/
143  CipInstance *first_instance = current_instance; /* allocate a block of memory for all created instances*/
144 
145  OPENER_ASSERT(NULL != current_instance);
146  /* fail if run out of memory */
147 
148  cip_class->number_of_instances += number_of_instances; /* add the number of instances just created to the total recorded by the class */
149 
150  for (int i = 0; i < number_of_instances; i++) /* initialize all the new instances */
151  {
152  *next_instance = current_instance; /* link the previous pointer to this new node */
153 
154  current_instance->instance_number = instance_number; /* assign the next sequential instance number */
155  current_instance->cip_class = cip_class; /* point each instance to its class */
156 
157  if (cip_class->number_of_attributes) /* if the class calls for instance attributes */
158  { /* then allocate storage for the attribute array */
159  current_instance->attributes = (CipAttributeStruct *) CipCalloc(
160  cip_class->number_of_attributes, sizeof(CipAttributeStruct) );
161  }
162 
163  next_instance = &current_instance->next; /* update pp to point to the next link of the current node */
164  instance_number++; /* update to the number of the next node*/
165  current_instance++; /* point to the next node in the calloc'ed array*/
166  }
167 
168  return first_instance;
169 }

298 void InsertAttribute(CipInstance *const instance,
299  const EipUint16 attribute_number,
300  const EipUint8 cip_type,
301  void *const data,
302  const EipByte cip_flags) {
303 
304  CipAttributeStruct *attribute = instance->attributes;
305  CipClass *class = instance->cip_class;
306 
307  OPENER_ASSERT(NULL != attribute);
308  /* adding a attribute to a class that was not declared to have any attributes is not allowed */
309  for (int i = 0; i < instance->cip_class->number_of_attributes; i++) {
310  if (attribute->data == NULL) { /* found non set attribute */
311  attribute->attribute_number = attribute_number;
312  attribute->type = cip_type;
313  attribute->attribute_flags = cip_flags;
314  attribute->data = data;
315 
316  OPENER_ASSERT(attribute_number <= class->highest_attribute_number);
317 
318  size_t index = CalculateIndex(attribute_number);
319 
320  class->get_single_bit_mask[index] |=
321  (cip_flags & kGetableSingle) ? 1 << (attribute_number) % 8 : 0;
322  class->get_all_bit_mask[index] |=
323  (cip_flags & kGetableAll) ? 1 << (attribute_number) % 8 : 0;
324  class->set_bit_mask[index] |= ( (cip_flags & kSetable) ? 1 : 0 )
325  << ( (attribute_number) % 8 );
326 
327  return;
328  }
329  attribute++;
330  }
331  OPENER_TRACE_ERR(
332  "Tried to insert to many attributes into class: %" PRIu32 ", instance %" PRIu32 "\n",
333  instance->cip_class->class_instance.instance_number,
334  instance->instance_number);
335  OPENER_ASSERT(0);
336  /* trying to insert too many attributes*/
337 }

443 int EncodeData(const EipUint8 cip_type,
444  const void *const cip_data,
445  EipUint8 **cip_message) {
446  int counter = 0;
447 
448  switch (cip_type)
449  /* check the data type of attribute */
450  {
451  case (kCipBool):
452  case (kCipSint):
453  case (kCipUsint):
454  case (kCipByte):
455  counter = AddSintToMessage(*(EipUint8 *) (cip_data), cip_message);
456  break;
457 
458  case (kCipInt):
459  case (kCipUint):
460  case (kCipWord):
461  counter = AddIntToMessage(*(EipUint16 *) (cip_data), cip_message);
462  break;
463 
464  case (kCipDint):
465  case (kCipUdint):
466  case (kCipDword):
467  case (kCipReal):
468  counter = AddDintToMessage(*(EipUint32 *) (cip_data), cip_message);
469  break;
470 
471 #ifdef OPENER_SUPPORT_64BIT_DATATYPES
472  case (kCipLint):
473  case (kCipUlint):
474  case (kCipLword):
475  case (kCipLreal):
476  counter = AddLintToMessage(*(EipUint64 *) (cip_data), cip_message);
477  break;
478 #endif
479 
480  case (kCipStime):
481  case (kCipDate):
482  case (kCipTimeOfDay):
483  case (kCipDateAndTime):
484  break;
485  case (kCipString): {
486  CipString *const string = (CipString *) cip_data;
487 
488  AddIntToMessage(*(EipUint16 *) &(string->length), cip_message);
489  memcpy(*cip_message, string->string, string->length);
490  *cip_message += string->length;
491 
492  counter = string->length + 2; /* we have a two byte length field */
493  if (counter & 0x01) {
494  /* we have an odd byte count */
495  **cip_message = 0;
496  ++(*cip_message);
497  counter++;
498  }
499  break;
500  }
501  case (kCipString2):
502  case (kCipFtime):
503  case (kCipLtime):
504  case (kCipItime):
505  case (kCipStringN):
506  break;
507 
508  case (kCipShortString): {
509  CipShortString *const short_string = (CipShortString *) cip_data;
510 
511  **cip_message = short_string->length;
512  ++(*cip_message);
513 
514  memcpy(*cip_message, short_string->string, short_string->length);
515  *cip_message += short_string->length;
516 
517  counter = short_string->length + 1;
518  break;
519  }
520 
521  case (kCipTime):
522  break;
523 
524  case (kCipEpath):
525  counter = EncodeEPath( (CipEpath *) cip_data, cip_message );
526  break;
527 
528  case (kCipEngUnit):
529  break;
530 
531  case (kCipUsintUsint): {
532  CipRevision *revision = (CipRevision *) cip_data;
533 
534  **cip_message = revision->major_revision;
535  ++(*cip_message);
536  **cip_message = revision->minor_revision;
537  ++(*cip_message);
538  counter = 2;
539  break;
540  }
541 
542  case (kCipUdintUdintUdintUdintUdintString): {
543  /* TCP/IP attribute 5 */
544  CipTcpIpNetworkInterfaceConfiguration *

546  (CipTcpIpNetworkInterfaceConfiguration *) cip_data;
547  counter += AddDintToMessage(
548  ntohl(tcp_ip_network_interface_configuration->ip_address),
549  cip_message);
550  counter += AddDintToMessage(
551  ntohl(tcp_ip_network_interface_configuration->network_mask),
552  cip_message);
553  counter += AddDintToMessage(
554  ntohl(tcp_ip_network_interface_configuration->gateway), cip_message);

565  }
566

569  memcpy(*cip_message, p, 6);

575  break;

671

742 int EncodeEPath(CipEpath *epath,
743  EipUint8 **message) {
744  unsigned int length = epath->path_size;
745  AddIntToMessage(epath->path_size, message);
746 
747  if (epath->class_id < 256) {
748  **message = 0x20; /*8Bit Class Id */
749  ++(*message);
750  **message = (EipUint8) epath->class_id;
751  ++(*message);
752  length -= 1;
753  } else {
754  **message = 0x21; /*16Bit Class Id */
755  ++(*message);
756  **message = 0; /*pad byte */
757  ++(*message);
758  AddIntToMessage(epath->class_id, message);
759  length -= 2;
760  }
761 
762  if (0 < length) {
763  if (epath->instance_number < 256) {
764  **message = 0x24; /*8Bit Instance Id */
765  ++(*message);
766  **message = (EipUint8) epath->instance_number;
767  ++(*message);
768  length -= 1;
769  } else {
770  **message = 0x25; /*16Bit Instance Id */
771  ++(*message);
772  **message = 0; /*padd byte */
773  ++(*message);
774  AddIntToMessage(epath->instance_number, message);
775  length -= 2;
776  }
777 
778  if (0 < length) {
779  if (epath->attribute_number < 256) {
780  **message = 0x30; /*8Bit Attribute Id */
781  ++(*message);
782  **message = (EipUint8) epath->attribute_number;
783  ++(*message);
784  length -= 1;
785  } else {
786  **message = 0x31; /*16Bit Attribute Id */
787  ++(*message);
788  **message = 0; /*pad byte */
789  ++(*message);
790  AddIntToMessage(epath->attribute_number, message);
791  length -= 2;
792  }
793  }
794  }
795 
796  return 2 + epath->path_size * 2; /* path size is in 16 bit chunks according to the specification */
797 }

799 int DecodePaddedEPath(CipEpath *epath,
800  const EipUint8 **message) {
801  unsigned int number_of_decoded_elements = 0;
802  const EipUint8 *message_runner = *message;
803 
804  epath->path_size = *message_runner;
805  message_runner++;
806  /* copy path to structure, in version 0.1 only 8 bit for Class,Instance and Attribute, need to be replaced with function */
807  epath->class_id = 0;
808  epath->instance_number = 0;
809  epath->attribute_number = 0;
810 
811  for (number_of_decoded_elements = 0;
812  number_of_decoded_elements < epath->path_size;
813  number_of_decoded_elements++) {
814  if ( kSegmentTypeReserved ==
815  ( (*message_runner) & kSegmentTypeReserved ) ) {
816  /* If invalid/reserved segment type, segment type greater than 0xE0 */
817  return kEipStatusError;
818  }
819 
820  switch (*message_runner) {
821  case SEGMENT_TYPE_LOGICAL_SEGMENT + LOGICAL_SEGMENT_TYPE_CLASS_ID +
822  LOGICAL_SEGMENT_FORMAT_EIGHT_BIT:
823  epath->class_id = *(EipUint8 *) (message_runner + 1);
824  message_runner += 2;
825  break;
826 
827  case SEGMENT_TYPE_LOGICAL_SEGMENT + LOGICAL_SEGMENT_TYPE_CLASS_ID +
828  LOGICAL_SEGMENT_FORMAT_SIXTEEN_BIT:
829  message_runner += 2;
830  epath->class_id = GetIntFromMessage( &(message_runner) );
831  number_of_decoded_elements++;
832  break;
833 
834  case SEGMENT_TYPE_LOGICAL_SEGMENT + LOGICAL_SEGMENT_TYPE_INSTANCE_ID +
835  LOGICAL_SEGMENT_FORMAT_EIGHT_BIT:
836  epath->instance_number = *(EipUint8 *) (message_runner + 1);
837  message_runner += 2;
838  break;
839 
840  case SEGMENT_TYPE_LOGICAL_SEGMENT + LOGICAL_SEGMENT_TYPE_INSTANCE_ID +
841  LOGICAL_SEGMENT_FORMAT_SIXTEEN_BIT:
842  message_runner += 2;
843  epath->instance_number = GetIntFromMessage( &(message_runner) );
844  number_of_decoded_elements++;
845  break;
846 
847  case SEGMENT_TYPE_LOGICAL_SEGMENT + LOGICAL_SEGMENT_TYPE_ATTRIBUTE_ID +
848  LOGICAL_SEGMENT_FORMAT_EIGHT_BIT:
849  epath->attribute_number = *(EipUint8 *) (message_runner + 1);
850  message_runner += 2;
851  break;
852 
853  case SEGMENT_TYPE_LOGICAL_SEGMENT + LOGICAL_SEGMENT_TYPE_ATTRIBUTE_ID +
854  LOGICAL_SEGMENT_FORMAT_SIXTEEN_BIT:
855  message_runner += 2;
856  epath->attribute_number = GetIntFromMessage( &(message_runner) );
857  number_of_decoded_elements++;
858  break;
859 
860  default:
861  OPENER_TRACE_ERR("wrong path requested\n");
862  return kEipStatusError;
863  break;
864  }
865  }
866 
867  *message = message_runner;
868  return number_of_decoded_elements * 2 + 1; /* number_of_decoded_elements times 2 as every encoding uses 2 bytes */
869 }
